جدول المصدر
Customers
مقاييس DAX — اضغط لعرض المعادلة
إجمالي العملاء COUNT
DAX
Total_Customers =
COUNT(Customers[AccountID])
عدد العدادات المتوقفة CALCULATE
DAX
Stopped_Meters =
CALCULATE(
    COUNT(Customers[MeterID]),
    Customers[MeterStatus] = "متوقف"
)
نسبة العدادات المتوقفة DIVIDE
DAX
Stopped_Meters_Rate =
DIVIDE(
    [Stopped_Meters],
    COUNT(Customers[MeterID]),
    0
)
نسبة العدادات الذكية DIVIDE
DAX
Smart_Meter_Rate =
DIVIDE(
    CALCULATE(
        COUNT(Customers[MeterID]),
        Customers[SmartMeterFlag] = "Y"
    ),
    COUNT(Customers[MeterID]),
    0
)
كشف الاستهلاك الشاذ — Box & Whisker لكل تصنيف عميل
الخطوة 1 — عمود محسوب في جدول Customers
-- نحسب استهلاك كل عميل مرة واحدة (عمود، ليس مقياساً)
-- RELATEDTABLE يجمع كل فواتير هذا العميل
CustomerTotalConsumption =
SUMX(
    RELATEDTABLE(Billing),
    Billing[Consumption_m3]
)

بعد إنشاء هذا العمود، كل المقاييس التالية تعمل عليه مباشرةً بدوال بسيطة — بلا CALCULATE أو سياق متقدّم.

الربيع الأول Q1 PERCENTILE
DAX
-- الربع الأدنى: 25% من العملاء تحت هذه القيمة
Consumption_Q1 =
PERCENTILE.INC(
    Customers[CustomerTotalConsumption],
    0.25
)
الوسيط Median PERCENTILE
DAX
-- الوسيط: القيمة الوسطى للاستهلاك
Consumption_Median =
MEDIAN(Customers[CustomerTotalConsumption])
الربيع الثالث Q3 PERCENTILE
DAX
-- الربع الأعلى: 75% من العملاء تحت هذه القيمة
Consumption_Q3 =
PERCENTILE.INC(
    Customers[CustomerTotalConsumption],
    0.75
)
المدى الربيعي IQR طرح
DAX
-- المدى الربيعي = Q3 ناقص Q1
Consumption_IQR =
[Consumption_Q3] - [Consumption_Q1]
الحد الأدنى (Lower Fence) طرح
DAX
-- أي عميل تحت هذا الحد = شاذ منخفض
Lower_Fence =
[Consumption_Q1] - 1.5 * [Consumption_IQR]
الحد الأعلى (Upper Fence) جمع
DAX
-- أي عميل فوق هذا الحد = شاذ مرتفع
Upper_Fence =
[Consumption_Q3] + 1.5 * [Consumption_IQR]
عدد العملاء الشاذين CALCULATE
DAX
-- نعدّ العملاء خارج الحدّين
Outlier_Customers =
VAR Low  = [Lower_Fence]
VAR High = [Upper_Fence]
RETURN
CALCULATE(
    COUNTROWS(Customers),
    Customers[CustomerTotalConsumption] < Low
        || Customers[CustomerTotalConsumption] > High
)
تصنيف العميل (عمود) IF
DAX — عمود محسوب اختياري
-- يلوّن النقاط الشاذة في الـ Box & Whisker
Outlier_Flag =
VAR c = Customers[CustomerTotalConsumption]
RETURN
IF(
    c < [Lower_Fence] || c > [Upper_Fence],
    "شاذ",
    "طبيعي"
)
المخططات البصرية
بطاقة KPI — إجمالي العملاء Gauge — نسبة المتوقفة Gauge — نسبة الذكية دائري — توزيع MeterStatus شريطي — العملاء حسب CustomerClass Box & Whisker — الاستهلاك الشاذ حسب التصنيف
